% #1 out, var name to save the result (1 or 0) to
% #2 scalar s, the first scalar
% #3 scalar t, the second scalar
\newcommand\equalsScalar[3][equalsScalarResult] {%
    \newScalar{#1}{1}
    \FPround{\@tmpValA}{#2}{9}
    \FPround{\@tmpValB}{#3}{9}
    \FPifeq{\@tmpValA}{\@tmpValB}
    \else
        \newScalar{#1}{0}
    \fi
}

% Compare two vectors for equality. Returns 1 if equal, 0 else.
% Example: \textbackslash newVec\{myNewVecA\}\{0.5, 1, 2\}
% Example: \textbackslash newVec\{myNewVecB\}\{0.5, 1, 2\}
% Example: \textbackslash equalsVec\{\textbackslash myNewVecA\}\{\textbackslash myNewVecB\}
% Example: \textbackslash equalsVecResult \(\mapsto{} 1\)
% #1 out, var name to save the result (1 or 0) to
% #2 vector v, the first vector
% #3 vector w, the second vector
\newcommand\equalsVec[3][equalsVecResult] {%
    \newScalar{#1}{1}
    \getVecHeight[@vecHeight]{#2}
    \edef\@sequence{\xintSeq[+1]{+1}{\@vecHeight}}%
    \xintFor* ##1 in \@sequence \do {%
        \FPround{\@tmpValA}{\xintNthElt{##1}{#2}}{9}
        \FPround{\@tmpValB}{\xintNthElt{##1}{#3}}{9}
        \FPifeq{\@tmpValA}{\@tmpValB}
        \else
            \newScalar{#1}{0}
        \fi
    }
}

% Can for example be used for Immutable Input Test
% #1 type, output type e.g. vec or scalar
% #2 vector v, the vector to check
% #3 vector c, the control vector to test against
% #4 string, context of what is tested
\newcommand\testMacroEquality[4] {%
    \csname equals#1\endcsname{#2}{#3}
    \FPifeq{\csname equals#1Result\endcsname}{1}
        \textcolor{teal}{$\checkmark$} equality test: #4
    \else
        \PackageWarning{glmatrix package tests}{Equality Test: Output does not match expectation}
        \textcolor{red}{$\times$} equality test: #4\quad
        result: \csname print#1\endcsname{#2}
        expected: \csname print#1\endcsname{#3}
    \fi
}

% #1 type, output type e.g. vec or scalar
% #2 macro, the macro to test
% #3 param 1, the first and only param to pass to the macro
% #4 vector c, the control vector to test against
% #5 string, context of what is tested
\newcommand\testMacroOnePar[5] {%
    \csname #2\endcsname[calculationResult]{#3}
    \csname equals#1\endcsname{\calculationResult}{#4}
    \FPifeq{\csname equals#1Result\endcsname}{1}
        \textcolor{teal}{$\checkmark$} #2 #5
    \else
        \PackageWarning{glmatrix package tests}{In #2: Output does not match expectation}
        \textcolor{red}{$\times$} #2 #5\quad
        result: \csname print#1\endcsname{\calculationResult}
        expected: \csname print#1\endcsname{#4}
        \calculationResult
        #4
    \fi
}
