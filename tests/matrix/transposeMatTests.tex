\subsubsection*{transposeMat} %
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% 1x1 matrix mixed sign
\newMat{matA}{{{5}}}
\newMat{matControl}{{{5}}}
\testMacroOnePar{Mat}{transposeMat}{\matA}{\matControl}{1x1 matrix mixed sign}

% 2x2 matrix mixed sign
\newMat{matA}{{{1,-2}{3,4}}}
\newMat{matControl}{{{1,3}{-2,4}}}
\testMacroOnePar{Mat}{transposeMat}{\matA}{\matControl}{2x2 matrix mixed sign}

% 2x3 matrix mixed sign
\newMat{matA}{{{1,-2,3.5}{4.5,0,-1}}}
\newMat{matControl}{{{1,4.5}{-2,0}{3.5,-1}}}
\testMacroOnePar{Mat}{transposeMat}{\matA}{\matControl}{2x3 matrix mixed sign}

% 3x2 matrix mixed sign
\newMat{matA}{{{1,-2}{3.5,4}{0,-1.5}}}
\newMat{matControl}{{{1,3.5,0}{-2,4,-1.5}}}
\testMacroOnePar{Mat}{transposeMat}{\matA}{\matControl}{3x2 matrix mixed sign}

% 3x3 matrix mixed sign
\newMat{matA}{{{1,-2,3}{4.5,0,-1}{2,-3.5,1}}}
\newMat{matControl}{{{1,4.5,2}{-2,0,-3.5}{3,-1,1}}}
\testMacroOnePar{Mat}{transposeMat}{\matA}{\matControl}{3x3 matrix mixed sign}

% 3x4 matrix mixed sign
\newMat{matA}{{{1,-2,3,0.5}{4.5,0,-1,2}{2,-3.5,1,-2}}}
\newMat{matControl}{{{1,4.5,2}{-2,0,-3.5}{3,-1,1}{0.5,2,-2}}}
\testMacroOnePar{Mat}{transposeMat}{\matA}{\matControl}{3x4 matrix mixed sign}

% 4x3 matrix mixed sign
\newMat{matA}{{{1,-2,3}{4.5,0,-1}{2,-3.5,1}{0.5,2,-2}}}
\newMat{matControl}{{{1,4.5,2,0.5}{-2,0,-3.5,2}{3,-1,1,-2}}}
\testMacroOnePar{Mat}{transposeMat}{\matA}{\matControl}{4x3 matrix mixed sign}

% 4x4 matrix mixed sign
\newMat{matA}{{{1,-2,3,0.5}{4.5,0,-1,2}{2,-3.5,1,-2}{0.5,2,-2,1}}}
\newMat{matControl}{{{1,4.5,2,0.5}{-2,0,-3.5,2}{3,-1,1,-2}{0.5,2,-2,1}}}
\testMacroOnePar{Mat}{transposeMat}{\matA}{\matControl}{4x4 matrix mixed sign}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% 2x2 matrix large integers
\newMat{matA}{{{1000000,-2000000}{3000000,4000000}}}
\newMat{matControl}{{{1000000,3000000}{-2000000,4000000}}}
\testMacroOnePar{Mat}{transposeMat}{\matA}{\matControl}{2x2 matrix large integers}

% 3x3 matrix large integers
\newMat{matA}{{{1000000,-2000000,3000000}{4500000,0,-1000000}{2000000,-3500000,1000000}}}
\newMat{matControl}{{{1000000,4500000,2000000}{-2000000,0,-3500000}{3000000,-1000000,1000000}}}
\testMacroOnePar{Mat}{transposeMat}{\matA}{\matControl}{3x3 matrix large integers}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% 2x2 matrix small floats
\newMat{matA}{{{0.000001,-0.000002}{0.000003,0.000004}}}
\newMat{matControl}{{{0.000001,0.000003}{-0.000002,0.000004}}}
\testMacroOnePar{Mat}{transposeMat}{\matA}{\matControl}{2x2 matrix small floats}

% 3x3 matrix small floats
\newMat{matA}{{{0.000001,-0.000002,0.000003}{0.0000045,0,-0.000001}{0.000002,-0.0000035,0.000001}}}
\newMat{matControl}{{{0.000001,0.0000045,0.000002}{-0.000002,0,-0.0000035}{0.000003,-0.000001,0.000001}}}
\testMacroOnePar{Mat}{transposeMat}{\matA}{\matControl}{3x3 matrix small floats}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% Immutable input
\newMat{matControl}{{{0.000001,-0.000002,0.000003}{0.0000045,0,-0.000001}{0.000002,-0.0000035,0.000001}}}
\testMacroEquality{Mat}{\matA}{\matControl}{immutable input}
