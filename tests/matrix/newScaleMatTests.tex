\newVec{vecA}{5}
\newScaleMat{matA}{\vecA}
\newMat{matControl}{{{5,0}{0,1}}}
\testMacroEquality{Mat}{\matA}{\matControl}{newScaleMat v1}

\newVec{vecA}{5, 2}
\newScaleMat{matA}{\vecA}
\newMat{matControl}{{{5,0,0}{0,2,0}{0,0,1}}}
\testMacroEquality{Mat}{\matA}{\matControl}{newScaleMat v2}

\newVec{vecA}{5, 2, 3}

\testMacroEquality{Mat}{\matA}{\matControl}{newScaleMat v4}
