% 1x1 matrix mixed sign
\newMat{matA}{{{5}}}
\newVec{vecB}{2}
\newVec{vecControl}{10}
\testMacroTwoPar{Vec}{mulMatVec}{\matA}{\vecB}{\vecControl}{1x1 matrix mixed sign}

% 2x2 matrix mixed sign
\newMat{matA}{{{1,2.0}{3,4}}}
\newVec{vecB}{1,3}
\newVec{vecControl}{7,15}
\testMacroTwoPar{Vec}{mulMatVec}{\matA}{\vecB}{\vecControl}{2x2 matrix mixed sign}

% 2x3 matrix mixed sign
\newMat{matA}{{{1,2,3}{4,5,6}}}

\newVec{vecControl}{17,36.5}

% 3x4 matrix mixed sign

\newVec{vecControl}{2,17.5,7.5,9}
