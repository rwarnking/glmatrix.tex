\subsubsection*{lenVec} %
%%%%%%%%%%%%%%%%%%%%%%%%%
% 1D vector mixed sign
\newVec{vecA}{5}
\newScalar{scalarControl}{5}
\testMacroOnePar{Scalar}{lenVec}{\vecA}{\scalarControl}{1D vector mixed sign}

% 2D vector mixed sign
\newVec{vecA}{5, -3}
\newScalar{scalarControl}{5.8309518948}
\testMacroOnePar{Scalar}{lenVec}{\vecA}{\scalarControl}{2D vector mixed sign}

% 3D vector mixed sign
\newVec{vecA}{1, -2, 3}
\newScalar{scalarControl}{3.7416573867}
\testMacroOnePar{Scalar}{lenVec}{\vecA}{\scalarControl}{3D vector mixed sign}

% 4D vector mixed sign, mixed type
\newVec{vecA}{1, -2.5, 3, -4}
\newScalar{scalarControl}{5.6789083458}
\testMacroOnePar{Scalar}{lenVec}{\vecA}{\scalarControl}{4D vector mixed sign, mixed type}

% 10D vector mixed sign, mixed type
\newVec{vecA}{1, -2, 3, -4, 5.5, -6.25, 7.75, -8.5, 9.125, -5.875}
\newScalar{scalarControl}{18.6924115619}
\testMacroOnePar{Scalar}{lenVec}{\vecA}{\scalarControl}{10D vector mixed sign, mixed type}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% 1D vector length 0
