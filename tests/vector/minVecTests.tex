\subsubsection*{minVec} %

% 2x 1D vector mixed sign
\newVec{vecA}{5}
\newVec{vecB}{-3}
\newVec{vecControl}{-3}
\testMacroTwoPar{Vec}{minVec}{\vecA}{\vecB}{\vecControl}{2x 1D vector mixed sign}

% 2x 2D vector mixed sign
\newVec{vecA}{5, -3}
\newVec{vecB}{-2, 4}
\newVec{vecControl}{-2, -3}
\testMacroTwoPar{Vec}{minVec}{\vecA}{\vecB}{\vecControl}{2x 2D vector mixed sign}

% 2x 3D vector mixed sign
\newVec{vecA}{1, -2, 3}
\newVec{vecB}{-4, 5, 6}
\newVec{vecControl}{-4, -2, 3}
\testMacroTwoPar{Vec}{minVec}{\vecA}{\vecB}{\vecControl}{2x 3D vector mixed sign}

% 2x 4D vector mixed sign
\newVec{vecA}{1, -2, 3, -4}
\newVec{vecB}{-5, 6, 7, -8}
\newVec{vecControl}{-5, -2, 3, -8}
\testMacroTwoPar{Vec}{minVec}{\vecA}{\vecB}{\vecControl}{2x 4D vector mixed sign}

% 2x 10D vector mixed sign, mixed type
\newVec{vecA}{1, -2, 3, -4, 5.5, -6.25, 7.75, -8.5, 9.125, -5.875}
\newVec{vecB}{-1, 2, 3, -4, -5.5, 6.00, 7.75, -8.5, -1.125, 10.275}
\newVec{vecControl}{-1, -2, 3, -4, -5.5, -6.25, 7.75, -8.5, -1.125, -5.875}
\testMacroTwoPar{Vec}{minVec}{\vecA}{\vecB}{\vecControl}{2x 10D vector mixed sign, mixed type}
