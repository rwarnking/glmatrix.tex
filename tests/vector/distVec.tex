\subsubsection*{distVec} %
%%%%%%%%%%%%%%%%%%%%%%%%%%
% 2x 1D vector mixed sign
\newVec{vecA}{5}
\newVec{vecB}{-3}
\newScalar{scalarControl}{8}
\testMacroTwoPar{Scalar}{distVec}{\vecA}{\vecB}{\scalarControl}{2x 1D vector mixed sign}

% 2x 2D vector mixed sign
\newVec{vecA}{5, -3}
\newVec{vecB}{-2, 4}
\newScalar{scalarControl}{9.8994949366}
\testMacroTwoPar{Scalar}{distVec}{\vecA}{\vecB}{\scalarControl}{2x 2D vector mixed sign}

% 2x 3D vector mixed sign
\newVec{vecA}{1, -2, 3}
\newVec{vecB}{-4, 5, 6}
\newScalar{scalarControl}{9.1104335791}
\testMacroTwoPar{Scalar}{distVec}{\vecA}{\vecB}{\scalarControl}{2x 3D vector mixed sign}

% 2x 4D vector mixed sign, mixed type
\newVec{vecA}{1, -2.5, 3, -4}
\newVec{vecB}{-5, 6, 7.25, -8}
\newScalar{scalarControl}{11.9294802904}
\testMacroTwoPar{Scalar}{distVec}{\vecA}{\vecB}{\scalarControl}{2x 4D vector mixed sign, mixed type}

% 2x 10D vector mixed sign, mixed type
\newVec{vecA}{1, -2, 3, -4, 5.5, -6.25, 7.75, -8.5, 9.125, -5.875}
\newVec{vecB}{-1, 2, 3, -4, -5.5, 6.00, 7.75, -8.5, -1.125, 10.275}
\newScalar{scalarControl}{25.6309871054}
\testMacroTwoPar{Scalar}{distVec}{\vecA}{\vecB}{\scalarControl}{2x 10D vector mixed sign, mixed type}

% Commutativity
\newVec{vecA}{1, -2, 3}
\newVec{vecB}{-4, 5, -6}
\distVec[calculationResultA]{\vecA}{\vecB}
\distVec[calculationResultB]{\vecB}{\vecA}
\testMacroEquality{Scalar}{\calculationResultA}{\calculationResultB}{commutativity}
