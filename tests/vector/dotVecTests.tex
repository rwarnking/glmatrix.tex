\subsubsection*{dotVec} %
%%%%%%%%%%%%%%%%%%%%%%%%%
% 2x 1D vector mixed sign
\newVec{vecA}{5}
\newVec{vecB}{-3}
\newScalar{scalarControl}{-15}
\testMacroTwoPar{Scalar}{dotVec}{\vecA}{\vecB}{\scalarControl}{2x 1D vector mixed sign}

% 2x 2D vector mixed sign
\newVec{vecA}{5, -3}
\newVec{vecB}{-2, 4}
\newScalar{scalarControl}{-22}
\testMacroTwoPar{Scalar}{dotVec}{\vecA}{\vecB}{\scalarControl}{2x 2D vector mixed sign}

% 2x 3D vector mixed sign
\newVec{vecA}{1, -2, 3}
\newVec{vecB}{-4, 5, 6}
\newScalar{scalarControl}{4}
\testMacroTwoPar{Scalar}{dotVec}{\vecA}{\vecB}{\scalarControl}{2x 3D vector mixed sign}

% 2x 4D vector mixed sign, mixed type
\newVec{vecA}{1, -2.5, 3, -4}
\newVec{vecB}{-5, 6, 7.25, -8}
\newScalar{scalarControl}{33.75}
\testMacroTwoPar{Scalar}{dotVec}{\vecA}{\vecB}{\scalarControl}{2x 4D vector mixed sign, mixed type}

% 2x 10D vector mixed sign, mixed type
\newVec{vecA}{1, -2, 3, -4, 5.5, -6.25, 7.75, -8.5, 9.125, -5.875}
\newVec{vecB}{-1, 2, 3, -4, -5.5, 6.00, 7.75, -8.5, -1.125, 10.275}
\newScalar{scalarControl}{13.93125}
\testMacroTwoPar{Scalar}{dotVec}{\vecA}{\vecB}{\scalarControl}{2x 10D vector mixed sign, mixed type}

% Immutable input
\newVec{vecControl}{0.00001, -0.00002, -0.00003}
\testMacroEquality{Vec}{\vecA}{\vecControl}{immutable input}

% Commutativity
\newVec{vecA}{1, -2, 3}
\newVec{vecB}{-4, 5, -6}
\dotVec[calculationResultA]{\vecA}{\vecB}
\dotVec[calculationResultB]{\vecB}{\vecA}
\testMacroEquality{Scalar}{\calculationResultA}{\calculationResultB}{commutativity}

% Self-dot gives squared magnitude
\newVec{vecA}{3, 4}
\newVec{vecB}{3, 4}
\newScalar{scalarControl}{25}
\testMacroTwoPar{Scalar}{dotVec}{\vecA}{\vecB}{\scalarControl}{2D vector self-dot gives squared magnitude}

% Orthogonal vectors
\newVec{vecA}{1, 0}
\newVec{vecB}{0, 1}
\newScalar{scalarControl}{0}
\testMacroTwoPar{Scalar}{dotVec}{\vecA}{\vecB}{\scalarControl}{2D vector orthogonal vectors}
