% 2x 3D vector mixed sign
\newVec{vecA}{1, -2, 3}
\newVec{vecB}{-4, 5, 6}
\testPrintOutputTwoPar{printCrossPComp}{vecA = (1, -2, 3), vecB = (-4, 5, 6)}{\vecA}{\vecB}{\begin{pmatrix}-2*6-3*5\\3*-4-1*6\\1*5--2*-4\end{pmatrix}}

% \newVec{vecA}{3, 4}

% 2x 3D vector cancellation
\newVec{vecA}{1, -2, 3}
\newVec{vecB}{2, -4, 6}
\testPrintOutputTwoPar{printCrossPComp}{vecA = (1, -2, 3), vecB = (2, -4, 6)}{\vecA}{\vecB}{\begin{pmatrix}-2*6-3*-4\\3*2-1*6\\1*-4--2*2\end{pmatrix}}

% \newVec{vecA}{100000, 200000}

% 2x 3D vector large integers
\newVec{vecA}{1000000, 2000000, 3000000}
\newVec{vecB}{3000000, 4000000, 5000000}
\testPrintOutputTwoPar{printCrossPComp}{vecA = (1000000, 2000000, 3000000), vecB = (3000000, 4000000, 5000000)}{\vecA}{\vecB}{\begin{pmatrix}2000000*5000000-3000000*4000000\\3000000*3000000-1000000*5000000\\1000000*4000000-2000000*3000000\end{pmatrix}}
