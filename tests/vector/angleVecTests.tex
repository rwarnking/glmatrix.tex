% 2x 1D vector mixed sign
\newVec{vecA}{5}
\newVec{vecB}{-3}
\newScalarPi{scalarControl}{1}
\testMacroTwoPar{Scalar}{angleVec}{\vecA}{\vecB}{\scalarControl}{2x 1D vector mixed sign}

% 2x 2D vector mixed sign
\newVec{vecA}{5, -3}
\newVec{vecB}{-2, 4}
\newScalar{scalarControl}{2.574863436066287}
\testMacroTwoPar{Scalar}{angleVec}{\vecA}{\vecB}{\scalarControl}{2x 2D vector mixed sign}

% 2x 3D vector mixed sign
\newVec{vecA}{1, -2, 3}
\newVec{vecB}{-4, 5, 6}
\newScalar{scalarControl}{1.448663944800604}
\testMacroTwoPar{Scalar}{angleVec}{\vecA}{\vecB}{\scalarControl}{2x 3D vector mixed sign}

% 2x 4D vector mixed sign, mixed type
\newVec{vecA}{1, -2.5, 3, -4}
\newVec{vecB}{-5, 6, 7.25, -8}
\newScalar{scalarControl}{1.108506896742926}
\testMacroTwoPar{Scalar}{angleVec}{\vecA}{\vecB}{\scalarControl}{2x 4D vector mixed sign, mixed type}

% 2x 10D vector mixed sign, mixed type
\newVec{vecA}{1, -2, 3, -4, 5.5, -6.25, 7.75, -8.5, 9.125, -5.875}
\newVec{vecB}{-1, 2, 3, -4, -5.5, 6.00, 7.75, -8.5, -1.125, 10.275}
\newScalar{scalarControl}{1.530090110279428}
\testMacroTwoPar{Scalar}{angleVec}{\vecA}{\vecB}{\scalarControl}{2x 10D vector mixed sign, mixed type}
