\subsubsection*{solveVecEQL} %
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% 2x 1D vector mixed sign
\newVec{vecA}{6}
\newVec{vecB}{-4}
\newScalar{scalarControl}{0.6666666666}
\testMacroTwoPar{Scalar}{solveVecEQL}{\vecA}{\vecB}{\scalarControl}{2x 1D vector mixed sign}

% 2x 2D vector mixed sign
\newVec{vecA}{3.3, -0.25}
\newVec{vecB}{-66, 5}
\newScalar{scalarControl}{20}
\testMacroTwoPar{Scalar}{solveVecEQL}{\vecA}{\vecB}{\scalarControl}{2x 2D vector mixed sign}

% 2x 3D vector mixed sign
\newVec{vecA}{1, 2, -3, 0.5}
\newVec{vecB}{-1.5, -3, 4.5, -0.75}
\newScalar{scalarControl}{1.5}
\testMacroTwoPar{Scalar}{solveVecEQL}{\vecA}{\vecB}{\scalarControl}{2x 3D vector mixed sign}

% 2x 4D vector mixed sign, mixed type
\newVec{vecA}{1, 2, -3, 0.5}
\newVec{vecB}{-4, -8, 12, -2.0}
\newScalar{scalarControl}{4}
\testMacroTwoPar{Scalar}{solveVecEQL}{\vecA}{\vecB}{\scalarControl}{2x 4D vector mixed sign, mixed type}

% 2x 10D vector mixed sign, mixed type
\newVec{vecA}{1, 2, -3, 0.5, 1.25, -0.8, 2.5, -1.7, 3.3, -0.25}
\newVec{vecB}{-2, -4, 6, -1.0, -2.5, 1.6, -5.0, 3.4, -6.6, 0.5}
\newScalar{scalarControl}{2}
\testMacroTwoPar{Scalar}{solveVecEQL}{\vecA}{\vecB}{\scalarControl}{2x 10D vector mixed sign, mixed type}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% 2x 1D vector cancellation

\testMacroEquality{Vec}{\vecA}{\vecControl}{immutable input}

% Commutativity
\newVec{vecA}{1, 2, -3, 0.5}
\newVec{vecB}{-1.5, -3, 4.5, -0.75}
\solveVecEQL[calculationResultA]{\vecA}{\vecB}
\solveVecEQL[calculationResultB]{\vecB}{\vecA}
\testMacroEquality{Scalar}{\calculationResultA}{\calculationResultB}{commutativity}
