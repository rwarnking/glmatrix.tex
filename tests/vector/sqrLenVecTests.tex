%%%%%%%%%%%%%%%%%%%%%%%%%
\subsubsection*{sqrLenVec} %
%%%%%%%%%%%%%%%%%%%%%%%%%
% 1D vector mixed sign
\newVec{vecA}{5}
\newScalar{scalarControl}{25}
\testMacroOnePar{Scalar}{sqrLenVec}{\vecA}{\scalarControl}{1D vector mixed sign}

% 2D vector mixed sign
\newVec{vecA}{5, -3}
\newScalar{scalarControl}{34}
\testMacroOnePar{Scalar}{sqrLenVec}{\vecA}{\scalarControl}{2D vector mixed sign}

% 3D vector mixed sign
\newVec{vecA}{1, -2, 3}
\newScalar{scalarControl}{14}
\testMacroOnePar{Scalar}{sqrLenVec}{\vecA}{\scalarControl}{3D vector mixed sign}

% 4D vector mixed sign, mixed type
\newVec{vecA}{1, -2.5, 3, -4}
\newScalar{scalarControl}{32.25}
\testMacroOnePar{Scalar}{sqrLenVec}{\vecA}{\scalarControl}{4D vector mixed sign, mixed type}

% 10D vector mixed sign, mixed type
\newVec{vecA}{1, -2, 3, -4, 5.5, -6.25, 7.75, -8.5, 9.125, -5.875}
\newScalar{scalarControl}{349.40625}
\testMacroOnePar{Scalar}{sqrLenVec}{\vecA}{\scalarControl}{10D vector mixed sign, mixed type}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% 1D vector length 0
\newVec{vecA}{0}
\newScalar{scalarControl}{0}
\testMacroOnePar{Scalar}{sqrLenVec}{\vecA}{\scalarControl}{1D vector length 0}

% 2D vector length 0
\newVec{vecA}{0, 0}
\newScalar{scalarControl}{0}
\testMacroOnePar{Scalar}{sqrLenVec}{\vecA}{\scalarControl}{2D vector length 0}

% 3D vector length 0

\newScalar{scalarControl}{0}
\testMacroOnePar{Scalar}{sqrLenVec}{\vecA}{\scalarControl}{3D vector length 0}

% 4D vector length 0

\newScalar{scalarControl}{0}
\testMacroOnePar{Scalar}{sqrLenVec}{\vecA}{\scalarControl}{4D vector length 0}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% 2D vector large integers
\newVec{vecA}{1000, 2000}
\newScalar{scalarControl}{5000000}
\testMacroOnePar{Scalar}{sqrLenVec}{\vecA}{\scalarControl}{2D vector large integers}

% 3D vector large integers
\newVec{vecA}{1000, 2000, 3000}
\newScalar{scalarControl}{14000000}
\testMacroOnePar{Scalar}{sqrLenVec}{\vecA}{\scalarControl}{3D vector large integers}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% 2D vector small floats
\newVec{vecA}{0.0001, -0.0002}
\newScalar{scalarControl}{0.00000005}
\testMacroOnePar{Scalar}{sqrLenVec}{\vecA}{\scalarControl}{2D vector small floats}

% 3D vector small floats
\newVec{vecA}{0.0001, -0.0002, -0.0003}
\newScalar{scalarControl}{0.00000014}
\testMacroOnePar{Scalar}{sqrLenVec}{\vecA}{\scalarControl}{3D vector small floats}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% Immutable input
\newVec{vecControl}{0.0001, -0.0002, -0.0003}
\testMacroEquality{Vec}{\vecA}{\vecControl}{immutable input}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% Sign invariance
\newVec{vecA}{3, 4}
\newVec{vecB}{-3, -4}
\sqrLenVec[calculationResultA]{\vecA}
\sqrLenVec[calculationResultB]{\vecB}
\testMacroEquality{Scalar}{\calculationResultA}{\calculationResultB}{sign invariance}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% Relation to dot product
\newVec{vecA}{3, 4}
\dotVec[calculationResultA]{\vecA}{\vecA}
\sqrLenVec[calculationResultB]{\vecA}
\testMacroEquality{Scalar}{\calculationResultA}{\calculationResultB}{relation to dot product}
