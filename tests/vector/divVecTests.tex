% 2x 1D vector mixed sign
% \newVec{vecA}{5}
% \newVec{vecB}{-2}
% \newVec{vecControl}{-2.5}
% \testMacroTwoPar{Vec}{divVec}{\vecA}{\vecB}{\vecControl}{2x 1D vector mixed sign}

% 2x 2D vector mixed sign
\newVec{vecA}{5, -3}
\newVec{vecB}{-2, 4}
\newVec{vecControl}{-2.5, -0.75}
\testMacroTwoPar{Vec}{divVec}{\vecA}{\vecB}{\vecControl}{2x 2D vector mixed sign}

% 2x 3D vector mixed sign
\newVec{vecA}{1, -2, 3}
\newVec{vecB}{-4, 5, 1.5}
\newVec{vecControl}{-0.25, -0.4, 2.0}
\testMacroTwoPar{Vec}{divVec}{\vecA}{\vecB}{\vecControl}{2x 3D vector mixed sign}

% 2x 4D vector mixed sign, mixed type
\newVec{vecA}{1, -2.5, 3, -4}
\newVec{vecB}{-0.5, 5, 1.5, -2}
\newVec{vecControl}{-2.0, -0.5, 2.0, 2.0}
\testMacroTwoPar{Vec}{divVec}{\vecA}{\vecB}{\vecControl}{2x 4D vector mixed sign, mixed type}

% 2x 10D vector mixed sign, mixed type
\newVec{vecA}{1, -2, 3.0, -4, 5, -6.0, 7, -8, 9.0, -10}
\newVec{vecB}{-1, 2, 1.5, -2, -5, 3, 7, -4, -3, 2}
\newVec{vecControl}{-1, -1, 2.0, 2.0, -1, -2.0, 1, 2, -3.0, -5.0}
\testMacroTwoPar{Vec}{divVec}{\vecA}{\vecB}{\vecControl}{2x 10D vector mixed sign, mixed type}
